```
def a():
    print(111)

def b():
    flag='od'
    print("test")
    yield 1
    flag='ida'
    print("nihao")
    yield 2
    print("wohao")
    print(flag)

f1=a()
f2=b()

print("before for loop")

for i in f2:
    print(i,type(i))
    print(666)

print("after for loop")
```
